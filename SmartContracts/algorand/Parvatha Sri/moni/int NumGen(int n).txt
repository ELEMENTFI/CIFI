int NumGen(int n) 
{ 

  

    // Iterate from 0 to N 

    for(int j = 1; j < n + 1; j++) 

    { 

  

        // Short-circuit operator is used  

        if (j % 5 == 0 || j % 7 == 0) 

            cout << j << " "; 

    } 

    return n; 
} 

  
// Driver code  

int main() 
{  

    // Input goes here  

    int N = 50; 

      

    // Iterating over generator function 

    NumGen(N); 

      

    return 0; 
}
